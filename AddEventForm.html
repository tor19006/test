    function submitForm() {


 // เรียกใช้ validateForm() เพื่อทำการตรวจสอบ

    if (!validateForm()) {

        return; // ถ้า validateForm() คืนค่า false จะไม่ส่งฟอร์ม

    }


  // แสดงแจ้งเตือนระหว่างประมวลผล

  swal({

    title: "Processing...",

    text: "Please wait while the task is being created.",

    icon: "info",

    buttons: false,

    closeOnClickOutside: false,

    closeOnEsc: false

  });


  var form = document.getElementById('taskForm');

  google.script.run.withSuccessHandler(function(response) {

    // แสดงแจ้งเตือนเมื่อบันทึกข้อมูลสำเร็จ

    swal({

      title: "Done!",

      text: response.message, // แสดงข้อความจาก response.message

      icon: "success",

      button: "เสร็จสิ้น"

    }).then(() => {

        // รีเซ็ตฟอร์มหลังจากปิด SweetAlert

       resetForm(); // รีเซ็ตฟอร์มที่นี่บนฝั่งไคลเอนต์


       // เรียกใช้ createProjectEvents() หลังจากการบันทึกเสร็จสิ้น

        google.script.run.createProjectEvents();

      });

    })

  .withFailureHandler(function(error) {

    // แสดงแจ้งเตือนเมื่อบันทึกไม่สำเร็จ

    swal({

      title: "Error",

      text: error.message, // ใช้ error.message เพื่อแสดงข้อความที่มาจากข้อผิดพลาด

      icon: "error",

      button: "เสร็จสิ้น"

    });

  })

  .addToSheet(Object.fromEntries(new FormData(form)));

}

    if (sheets) sheets.value = "";

    if (statusRadios) {

        statusRadios.forEach(radio => radio.checked = false);

    }

    if (typeRadios) {

        typeRadios.forEach(radio => radio.checked = false);

    }

    if (note) note.value = "";


    // รีเซ็ตฟิลด์ที่ถูกสร้างด้วย updateFormFields()

    if (additionalFields) additionalFields.innerHTML = "";
